1. Policy Rule (YAML) 
# Procurement Policy v2.1
# Rule: Check Price vs Contract

- rule_id: "PRICE_VARIANCE"
  target: "unit_price"
  compare_with: "contract_price"
  threshold:
    max_variance_pct: 5.0  # Max 5%
    action: "FLAG_ANOMALY"
  
- rule_id: "NO_PO_AFTER_INVOICE"
  enabled: true
